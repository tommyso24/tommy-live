```javascript
(function() {
  // Stars animation
  const starsEl = document.getElementById('stars');
  for (let i = 0; i < 45; i++) {
    const s = document.createElement('div');
    s.className = 'star';
    s.style.left = Math.random() * 100 + '%';
    s.style.top = Math.random() * 100 + '%';
    s.style.setProperty('--dur', (1.5 + Math.random() * 2.5) + 's');
    s.style.animationDelay = Math.random() * 2 + 's';
    starsEl.appendChild(s);
  }

  // State
  let people = [];
  let expenses = [];
  let currentStep = 1;

  // DOM refs
  const personName = document.getElementById('personName');
  const btnAddPerson = document.getElementById('btnAddPerson');
  const personTags = document.getElementById('personTags');
  const personCount = document.getElementById('personCount');
  const btnToStep2 = document.getElementById('btnToStep2');

  // Step navigation
  function showStep(step) {
    currentStep = step;
    document.getElementById('step1').style.display = step === 1 ? 'block' : 'none';
    document.getElementById('step2').style.display = step === 2 ? 'block' : 'none';
    document.getElementById('step3').style.display = step === 3 ? 'block' : 'none';

    for (let i = 1; i <= 3; i++) {
      const dot = document.getElementById('stepDot' + i);
      dot.classList.remove('active', 'done');
      if (i < step) dot.classList.add('done');
      else if (i === step) dot.classList.add('active');
    }
    for (let i = 1; i <= 2; i++) {
      const line = document.getElementById('stepLine' + i);
      line.classList.toggle('done', i < step);
    }
  }

  // People management
  function addPerson() {
    const name = personName.value.trim();
    if (!name || name.length > 10) return;
    if (people.includes(name)) {
      personName.value = '';
      return;
    }
    people.push(name);
    personName.value = '';
    personName.focus();
    renderPeople();
  }

  function removePerson(name) {
    people = people.filter(p => p !== name);
    // Also remove from any expenses that reference this person
    expenses.forEach(exp => {
      if (exp.splitType === 'custom') {
        exp.splitPeople = exp.splitPeople.filter(p => p !== name);
      }
    });
    renderPeople();
  }

  function renderPeople() {
    personTags.innerHTML = '';
    people.forEach(name => {
      const tag = document.createElement('div');
      tag.className = 'tag';
      tag.innerHTML = `${escapeHTML(name)} <span class="remove" data-name="${escapeHTML(name)}">×</span>`;
      tag.querySelector('.remove').onclick = () => removePerson(name);
      personTags.appendChild(tag);
    });
    personCount.textContent = people.length > 0 ? `已添加 ${people.length} 人` : '';
    btnToStep2.disabled = people.length < 2;
  }

  function escapeHTML(str) {
    const p = document.createElement('p');
    p.textContent = str;
    return p.innerHTML;
  }

  btnAddPerson.addEventListener('click', addPerson);
  personName.addEventListener('keydown', e => { if (e.key === 'Enter') addPerson(); });
  btnToStep2.addEventListener('click', () => { showStep(2); renderPeopleChecks(); });
```

```javascript
    splitAll.classList.remove('active');
    peopleChecks.style.display = 'flex';

  function renderPeopleChecks() {
    peopleChecks.innerHTML = '';
    people.forEach(name => {
      const label = document.createElement('label');
      const cb = document.createElement('input');
      cb.type = 'checkbox';
      cb.checked = true;
      cb.addEventListener('change', () => {
        label.classList.toggle('checked', cb.checked);
      });
      label.className = 'checked';
      label.appendChild(cb);
      label.appendChild(document.createTextNode(name));
      peopleChecks.appendChild(label);
    });
  }

  function addExpense() {
    const name = expenseName.value.trim();
    const amount = parseFloat(expenseAmount.value);
    if (!name || name.length > 20) return;
    if (!amount || amount < 0.01 || amount > 99999) return;

    let splitPeople = [];
    if (currentSplitType === 'custom') {
      const checks = peopleChecks.querySelectorAll('input[type="checkbox"]');
      checks.forEach((cb, i) => { if (cb.checked) splitPeople.push(people[i]); });
      if (splitPeople.length === 0) return;
    }

    expenses.push({
      name,
      amount: Math.round(amount * 100) / 100,
      splitType: currentSplitType,
      splitPeople: currentSplitType === 'all' ? [...people] : splitPeople
    });

    expenseName.value = '';
    expenseAmount.value = '';
    expenseName.focus();
    renderExpenses();
  }

  function removeExpense(index) {
    expenses.splice(index, 1);
    renderExpenses();
  }

  function renderExpenses() {
    expenseList.innerHTML = '';
    let total = 0;

    expenses.forEach((exp, idx) => {
      total += exp.amount;
      const div = document.createElement('div');
      div.className = 'expense-item';
      const splitText = exp.splitType === 'all'
        ? '所有人均分'
        : exp.splitPeople.join('、');
      div.innerHTML = `
        <div class="info">
          <div class="name">${escapeHTML(exp.name)}</div>
          <div class="split-info">${escapeHTML(splitText)}</div>
        </div>
        <div class="amount">¥${exp.amount.toFixed(2)}</div>
        <span class="remove-expense" data-idx="${idx}">×</span>
      `;
      div.querySelector('.remove-expense').onclick = () => removeExpense(idx);
      expenseList.appendChild(div);
    });

    if (expenses.length > 0) {
      expenseTotal.style.display = 'block';
      expenseTotal.textContent = `合计：¥${total.toFixed(2)}`;
    } else {
      expenseTotal.style.display = 'none';
    }

    btnCalculate.disabled = expenses.length === 0;
  }

  btnAddExpense.addEventListener('click', addExpense);
  expenseAmount.addEventListener('keydown', e => { if (e.key === 'Enter') addExpense(); });
  btnBackTo1.addEventListener('click', () => showStep(1));
  btnCalculate.addEventListener('click', () => { calculate(); showStep(3); });
```

```javascript
  // Step 3 DOM refs
  const resultCards = document.getElementById('resultCards');
  const payerSelect = document.getElementById('payerSelect');
  const transferList = document.getElementById('transferList');
  const btnBackTo2 = document.getElementById('btnBackTo2');
  const btnRestart = document.getElementById('btnRestart');

  let personTotals = {}; // { name: totalOwed }

  function calculate() {
    personTotals = {};
    people.forEach(name => { personTotals[name] = 0; });

    // Per-person detail tracking
    const personDetails = {};
    people.forEach(name => { personDetails[name] = []; });

    expenses.forEach(exp => {
      const splitPeople = exp.splitType === 'all' ? [...people] : exp.splitPeople;
      const count = splitPeople.length;
      if (count === 0) return;

      const perPerson = Math.floor(exp.amount * 100 / count) / 100;
      const remainder = Math.round(exp.amount * 100 - perPerson * 100 * count);

      splitPeople.forEach((name, i) => {
        let share = perPerson;
        if (i === 0 && remainder > 0) {
          share = Math.round((perPerson * 100 + remainder)) / 100;
        }
        personTotals[name] = Math.round((personTotals[name] + share) * 100) / 100;
        personDetails[name].push({ name: exp.name, amount: share });
      });
    });

    // Render result cards
    resultCards.innerHTML = '';
    people.forEach(name => {
      const card = document.createElement('div');
      card.className = 'result-card';
      const items = personDetails[name].map(
        d => `${escapeHTML(d.name)}: ¥${d.amount.toFixed(2)}`
      ).join('<br>');
      card.innerHTML = `
        <div class="person-name">${escapeHTML(name)}</div>
        <div class="person-items">${items}</div>
        <div class="person-total">应付：¥${personTotals[name].toFixed(2)}</div>
      `;
      resultCards.appendChild(card);
    });

    // Payer select
    payerSelect.innerHTML = '<option value="">请选择垫付人</option>';
    people.forEach(name => {
      const opt = document.createElement('option');
      opt.value = name;
      opt.textContent = name;
      payerSelect.appendChild(opt);
    });

    transferList.innerHTML = '<p style="color: var(--text-dim); text-align: center;">选择垫付人后显示转账方案</p>';
  }

  payerSelect.addEventListener('change', () => {
    const payer = payerSelect.value;
    if (!payer) {
      transferList.innerHTML = '<p style="color: var(--text-dim); text-align: center;">选择垫付人后显示转账方案</p>';
      return;
    }

    transferList.innerHTML = '';
    let hasTransfer = false;

    people.forEach(name => {
      if (name === payer) return;
      const amount = personTotals[name];
      if (amount <= 0) return;
      hasTransfer = true;

      const div = document.createElement('div');
      div.className = 'transfer-item';
      div.innerHTML = `${escapeHTML(name)} <span class="arrow">→</span> ${escapeHTML(payer)}：<span class="transfer-amount">¥${amount.toFixed(2)}</span>`;
      transferList.appendChild(div);
    });

    if (!hasTransfer) {
      transferList.innerHTML = '<p style="color: var(--text-dim); text-align: center;">无需转账</p>';
    }
  });

  btnBackTo2.addEventListener('click', () => showStep(2));
  btnRestart.addEventListener('click', () => {
    people = [];
    expenses = [];
    currentSplitType = 'all';
    splitAll.classList.add('active');
    splitCustom.classList.remove('active');
    peopleChecks.style.display = 'none';
    renderPeople();
    renderExpenses();
    showStep(1);
  });

  // Init
  showStep(1);
})();
```
